      const id = req.params.id;
      const jobData = req.body;
      const updatedData = {
        $set: jobData,
      }
      const query = { _id: new ObjectId(id) };
      const option = { upsert: true }
      const result = await jobsCollection.updateOne(query, updatedData, option);
      res.send(result);
    })
